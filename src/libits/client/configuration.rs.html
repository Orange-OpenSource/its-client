67impl Configuration {
68    #[cfg(feature = "mobility")]
69    pub fn component_name(&self, modifier: Option<u32>) -> String {
70        let station_id: String = match &self.node {
71            Some(node_configuration) => node_configuration
72                .read()
73                .unwrap()
74                .station_id(modifier)
75                .to_string(),
76            None => self.mobility.station_id.clone(),
77        };
78        format!("{}_{}", self.mqtt.mqtt_options.client_id(), station_id)
79    }
80
81    #[cfg(feature = "mobility")]
82    pub fn set_node_configuration(&mut self, node_configuration: NodeConfiguration) {
83        self.node = Some(RwLock::new(node_configuration));
84    }
85
86    pub fn set_mqtt_credentials(&mut self, username: &str, password: &str) {
87        self.mqtt.mqtt_options.set_credentials(username, password);
88    }
89
90    pub fn get<T: FromStr>(
91        &self,
92        section: Option<&'static str>,
93        key: &'static str,
94    ) -> Result<T, ConfigurationError> {
95        if self.custom_settings.is_some() {
96            match get_optional_field(section, key, self.custom_settings.as_ref().unwrap()) {
97                Ok(result) => match result {
98                    Some(value) => Ok(value),
99                    _ => Err(FieldNotFound(key)),
100                },
101                Err(e) => Err(e),
102            }
103        } else {
104            Err(NoCustomSettings)
105        }
106    }
107
108    pub fn set<T: Into<String>>(&mut self, section: Option<&str>, key: &str, value: T) {
109        if self.custom_settings.is_none() {
110            self.custom_settings = Some(Ini::default())
111        }
112        self.custom_settings
113            .as_mut()
114            .unwrap()
115            .with_section(section)
116            .set(key, value);
117    }
118}

120pub(crate) fn get_optional_field<T: FromStr>(
121    section: Option<&'static str>,
122    field: &'static str,
123    ini_config: &Ini,
124) -> Result<Option<T>, ConfigurationError> {
125    let section = if let Some(section) = ini_config.section(section) {
126        section
127    } else {
128        ini_config.general_section()
129    };
130    get_optional_from_section(field, section)
131}
132
133pub(crate) fn get_optional_from_section<T: FromStr>(
134    field: &'static str,
135    properties: &Properties,
136) -> Result<Option<T>, ConfigurationError> {
137    if let Some(value) = properties.get(field) {
138        match T::from_str(value) {
139            Ok(value) => Ok(Some(value)),
140            Err(_) => Err(TypeError(field, type_name::<T>())),
141        }
142    } else {
143        Ok(None)
144    }
145}
146
147pub(crate) fn get_mandatory_field<T: FromStr>(
148    section: Option<&'static str>,
149    field: &'static str,
150    ini_config: &Ini,
151) -> Result<T, ConfigurationError> {
152    if let Some(properties) = ini_config.section(section) {
153        if let Some(value) = properties.get(field) {
154            match T::from_str(value) {
155                Ok(value) => Ok(value),
156                Err(_e) => Err(TypeError(field, type_name::<T>())),
157            }
158        } else {
159            Err(MissingMandatoryField(field, section.unwrap_or_default()))
160        }
161    } else {
162        Err(MissingMandatorySection(section.unwrap_or_default()))
163    }
164}
165
166pub(crate) fn get_mandatory_from_section<T: FromStr>(
167    field: &'static str,
168    section: (&'static str, &Properties),
169) -> Result<T, ConfigurationError> {
170    match section.1.get(field) {
171        Some(value) => match T::from_str(value) {
172            Ok(value) => Ok(value),
173            Err(_e) => Err(TypeError(field, type_name::<T>())),
174        },
175        None => Err(MissingMandatoryField(field, section.0)),
176    }
177}
178
179pub(crate) fn pick_mandatory_section(
180    section: &'static str,
181    ini_config: &mut Ini,
182) -> Result<Properties, ConfigurationError> {
183    match ini_config.delete(Some(section)) {
184        Some(properties) => Ok(properties),
185        None => Err(MissingMandatorySection(section)),
186    }
187}

189impl TryFrom<Ini> for Configuration {
190    type Error = ConfigurationError;
191
192    fn try_from(ini_config: Ini) -> Result<Self, Self::Error> {
193        let mut ini_config = ini_config;
194
195        Ok(Configuration {
196            mqtt: MqttConfiguration::try_from(&pick_mandatory_section(
197                MQTT_SECTION,
198                &mut ini_config,
199            )?)?,
200            #[cfg(feature = "geo_routing")]
201            geo: GeoConfiguration::try_from(&pick_mandatory_section(
202                GEO_SECTION,
203                &mut ini_config,
204            )?)?,
205            #[cfg(feature = "telemetry")]
206            telemetry: TelemetryConfiguration::try_from(&pick_mandatory_section(
207                TELEMETRY_SECTION,
208                &mut ini_config,
209            )?)?,
210            #[cfg(feature = "mobility")]
211            mobility: MobilityConfiguration::try_from(&pick_mandatory_section(
212                STATION_SECTION,
213                &mut ini_config,
214            )?)?,
215            #[cfg(feature = "mobility")]
216            node: match ini_config.section(Some(NODE_SECTION)) {
217                Some(properties) => Some(RwLock::new(NodeConfiguration::try_from(properties)?)),
218                None => None,
219            },
220            custom_settings: Some(ini_config),
221        })
222    }
223}

225/// Creates a logger that outputs to stdout.
226///
227/// This function initializes a logger using the `flexi_logger` crate, which logs messages to the standard output (stdout).
228/// The logger's log level is set based on the environment variable or defaults to "info".
229///
230/// # Returns
231///
232/// A `Result` containing a `flexi_logger::LoggerHandle` if the logger is successfully initialized, or a boxed `dyn std::error::Error` if an error occurs.
233///
234/// # Errors
235///
236/// This function will return an error if the logger fails to initialize.
237///
238/// # Examples
239///
240/// ```rust
241/// use libits::client::configuration::create_stdout_logger;
242/// let _logger = create_stdout_logger().expect("Logger initialization failed");
243/// ```
244///
245/// # Dependencies
246///
247/// This function requires the `flexi_logger` and `log` crates.
248///
249/// # Notes
250///
251/// The logger is configured to print messages to stdout and includes a message indicating that the logger is ready.
252///
253/// # See Also
254///
255/// - `flexi_logger::Logger`
256/// - `log::info`
257pub fn create_stdout_logger() -> Result<flexi_logger::LoggerHandle, Box<dyn std::error::Error>> {
258    let logger = Logger::try_with_env_or_str("info")?
259        .log_to_stdout()
260        .print_message()
261        .start()?;
262    info!("Logger ready on stdout");
263    Ok(logger)
264}
